    <script>
        function generatePassword() {
            const length = document.getElementById('passLength').value;
            const includeNumbers = document.getElementById('numOpt').checked;
            const includeSymbols = document.getElementById('symOpt').checked;

            let charSet = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
            if (includeNumbers) charSet += "0123456789";
            if (includeSymbols) charSet += "!@#$%^&*()_+~`|}{[]:;?><,./-=";

            let password = "";
            for (let i = 0; i < length; i++) {
                const randomIndex = Math.floor(Math.random() * charSet.length);
                password += charSet[randomIndex];
            }

            document.getElementById('passDisplay').innerText = password;
        }
        function copyPass() {
            const pass = document.getElementById('passDisplay').innerText;
            if(pass === "اضغط على توليد") return;
            navigator.clipboard.writeText(pass);
            alert("تم نسخ كلمة المرور بنجاح!");
        }
    </script>
